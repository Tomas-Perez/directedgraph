\section{Assignment}
  \begin{enumerate}
    \item Specify a directed non-weighted graph.
    \item Implement and test a directed non-weighted graph with an edges linked list. Calculate the order of each operation.
    \item To test the correct functionality of the previous class:
      \begin{enumerate}[label=(\roman*)]
        \item Implement a function to load data to a graph (for both vertices and edges).
        \item Implement a function to generate a random graph.
        \item Implement a function to show the graph on screen (either a list with vertices and edges or a graph drawing).
      \end{enumerate}
    \item Implement all three traversal algorithms (Plain search, BFS and DFS)
    \item Modifying previous functions, write and test algorithms that:
      \begin{enumerate}[label=(\roman*)]
        \item Calculate the amount of sources and sinks.
        \item Verify if the graph is weakly connected.
        \item Given two vertices, verify if a path of length 2 exists between them.
        \item Implement Warshall's algorithm.
      \end{enumerate}
  \end{enumerate}

  \subsection{Graph Related Algorithms}
    \textit{Description:}
      Algorithms related to graph theory

      \methodSpec{getSources}
      {Graph}
      {Number}
      {Given a graph returns the number of source vertices.}
      {Graph must exist}
      {Return the amount of source vertices in the graph}
      {Analyzer}

      \methodSpec{getSinks}
      {Graph}
      {Number}
      {Given a graph returns the number of sink vertices.}
      {Graph must exist}
      {Return the amount of sink vertices in the graph}
      {Analyzer}

      \methodSpec{isWeaklyConnected}
      {Graph}
      {Boolean}
      {Given a graph, checks whether it is weakly connected.}
      {Graph must exist}
      {
        \begin{itemize}
          \item True if the graph is weakly connected.
          \item False if the graph is not weakly connected.
        \end{itemize}
      }
      {Analyzer}

      \methodSpec{pathOfLengthTwoExists}
      {Graph x Vertex x Vertex}
      {Boolean}
      {Given a graph and two vertices checks if a path of length 2 exists between them in the graph.}
      {Graph and both vertices must exist}
      {
        \begin{itemize}
          \item True if a path of length two between the two vertces exists.
          \item False if a path of length two between the two vertces does not exist.
        \end{itemize}
      }
      {Analyzer}

      \methodSpec{warshall}
      {Graph}
      {Transitive Closure}
      {Given a graph calculates its transitive closure using Warshall's.}
      {Graph must exist}
      {The transitive closure must have the capability of calculating whether
      there is a path between two vertices in O(1) time.}
      {Analyzer}
